/* FALTA ADICIONAR OS TIPOS DAS VARI√ÅVEIS PRINCIPAIS E SUAS CONSTRAINTS EM CADA TABELA */


create schema T1BD;


create table T1BD.paises (
	sigla char(3),
	nome_extenso varchar(30),
	
	constraint pk_paises primary key(sigla),
	constraint uk_paises unique(nome_extenso)
);


create table T1BD.pib_total (
	ano date,
	sigla char(3),
	pib_total ...,
	
	constraint pk_pib_total primary key(ano, sigla),
	constraint fk_pib_total foreign key(ano)
		references(paises) on delete cascade,
	
	constraint ...
);
